Icon Component

The uilibrary-icon component provides a flexible way to display icons with multiple sizes, appearances, and styling options throughout your application.

Standalone use

The Icon Component is dependant on AngularSvgIconModule which requires different setup for NgModules use vs Standalone use.

For NgModules use simply import the IconModule as normal. For Standalone use register the IconModule.forStandalone() and import the IconComponent in your Standalone components.

        
          import { IconModule } from '@conferma/uilibrary'

          export const appConfig: ApplicationConfig = {
            providers: [
              IconModule.forStandalone(),
              // more providers
            ]
          }
        
      
        
          import { IconComponent } from '@conferma/uilibrary';

          @Component({
            selector: 'app-my-component',
            imports: [IconComponent],
            templateUrl: './my-component.html',
            styleUrl: './my-component.scss',
            standalone: true // enabled is the default in most recent versions of Angular.
          })
          export class MyComponent { }
